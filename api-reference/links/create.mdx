---
title: Create a link
description: Create a link for the authenticated workspace.
---

## Authentication

This endpoint requires authentication using a Bearer token.

```bash
Authorization: Bearer <token>
```

## Endpoint

<ParamField path="method" type="string" required>
  POST
</ParamField>

<ParamField path="endpoint" type="string" required>
  /links
</ParamField>

## Request Body

<ParamField body="url" type="string" required>
  The destination URL of the short link.
  
  **Example:** `https://google.com`
</ParamField>

<ParamField body="domain" type="string">
  The domain of the short link (without protocol). If not provided, the primary domain for the workspace will be used (or `dub.sh` if the workspace has no domains).
</ParamField>

<ParamField body="key" type="string">
  The short link slug. If not provided, a random 7-character slug will be generated.
</ParamField>

<ParamField body="keyLength" type="number">
  The length of the short link slug. Defaults to 7 if not provided. When used with `prefix`, the total length of the key will be `prefix.length + keyLength`.
  
  **Min:** 3  
  **Max:** 190
</ParamField>

<ParamField body="externalId" type="string">
  The ID of the link in your database. If set, it can be used to identify the link in future API requests (must be prefixed with 'ext_' when passed as a query parameter). This key is unique across your workspace.
  
  **Example:** `123456`
</ParamField>

<ParamField body="tenantId" type="string">
  The ID of the tenant that created the link inside your system. If set, it can be used to fetch all links for a tenant.
</ParamField>

<ParamField body="programId" type="string">
  The ID of the program the short link is associated with.
</ParamField>

<ParamField body="partnerId" type="string">
  The ID of the partner the short link is associated with.
</ParamField>

<ParamField body="prefix" type="string">
  The prefix of the short link slug for randomly-generated keys (e.g. if prefix is `/c/`, generated keys will be in the `/c/:key` format). Will be ignored if `key` is provided.
</ParamField>

<ParamField body="trackConversion" type="boolean">
  Whether to track conversions for the short link. Defaults to `false` if not provided.
</ParamField>

<ParamField body="archived" type="boolean">
  Whether the short link is archived. Defaults to `false` if not provided.
</ParamField>

<ParamField body="tagIds" type="array">
  The unique IDs of the tags assigned to the short link.
  
  **Example:** `["clux0rgak00011..."]`
</ParamField>

<ParamField body="tagNames" type="array">
  The unique name of the tags assigned to the short link (case insensitive).
</ParamField>

<ParamField body="folderId" type="string">
  The unique ID existing folder to assign the short link to.
</ParamField>

<ParamField body="comments" type="string">
  The comments for the short link.
</ParamField>

<ParamField body="expiresAt" type="string">
  The date and time when the short link will expire at.
</ParamField>

<ParamField body="expiredUrl" type="string">
  The URL to redirect to when the short link has expired.
</ParamField>

<ParamField body="password" type="string">
  The password required to access the destination URL of the short link.
</ParamField>

<ParamField body="proxy" type="boolean">
  Whether the short link uses Custom Link Previews feature. Defaults to `false` if not provided.
</ParamField>

<ParamField body="title" type="string">
  The custom link preview title (og:title). Will be used for Custom Link Previews if `proxy` is true. Learn more: https://d.to/og
</ParamField>

<ParamField body="description" type="string">
  The custom link preview description (og:description). Will be used for Custom Link Previews if `proxy` is true. Learn more: https://d.to/og
</ParamField>

<ParamField body="image" type="string">
  The custom link preview image (og:image). Will be used for Custom Link Previews if `proxy` is true. Learn more: https://d.to/og
</ParamField>

<ParamField body="video" type="string">
  The custom link preview video (og:video). Will be used for Custom Link Previews if `proxy` is true. Learn more: https://d.to/og
</ParamField>

<ParamField body="rewrite" type="boolean">
  Whether the short link uses link cloaking. Defaults to `false` if not provided.
</ParamField>

<ParamField body="ios" type="string">
  The iOS destination URL for the short link for iOS device targeting.
</ParamField>

<ParamField body="android" type="string">
  The Android destination URL for the short link for Android device targeting.
</ParamField>

<ParamField body="geo" type="object">
  Geo targeting information for the short link in JSON format `{[COUNTRY]: https://example.com }`. See https://d.to/geo for more information.
</ParamField>

<ParamField body="doIndex" type="boolean">
  Allow search engines to index your short link. Defaults to `false` if not provided. Learn more: https://d.to/noindex
</ParamField>

<ParamField body="utm_source" type="string">
  The UTM source of the short link. If set, this will populate or override the UTM source in the destination URL.
</ParamField>

<ParamField body="utm_medium" type="string">
  The UTM medium of the short link. If set, this will populate or override the UTM medium in the destination URL.
</ParamField>

<ParamField body="utm_campaign" type="string">
  The UTM campaign of the short link. If set, this will populate or override the UTM campaign in the destination URL.
</ParamField>

<ParamField body="utm_term" type="string">
  The UTM term of the short link. If set, this will populate or override the UTM term in the destination URL.
</ParamField>

<ParamField body="utm_content" type="string">
  The UTM content of the short link. If set, this will populate or override the UTM content in the destination URL.
</ParamField>

<ParamField body="ref" type="string">
  The referral tag of the short link. If set, this will populate or override the `ref` query parameter in the destination URL.
</ParamField>

<ParamField body="webhookIds" type="array">
  An array of webhook IDs to trigger when the link is clicked. These webhooks will receive click event data.
</ParamField>

<ParamField body="testVariants" type="array">
  An array of A/B test URLs and the percentage of traffic to send to each URL.
  
  **Example:**
  ```json
  [
    {
      "url": "https://example.com/variant-1",
      "percentage": 50
    },
    {
      "url": "https://example.com/variant-2",
      "percentage": 50
    }
  ]
  ```
</ParamField>

<ParamField body="testStartedAt" type="string">
  The date and time when the tests started.
</ParamField>

<ParamField body="testCompletedAt" type="string">
  The date and time when the tests were or will be completed.
</ParamField>

```bash cURL
  -H "Content-Type: application/json" \
  -d '{

    "domain": "dub.sh",
    "key": "try"
  }'
```

```javascript Node.js
const response = await fetch('https://api.dub.co/links', {

  },
  body: JSON.stringify({

    key: 'try'
  })
});
```
