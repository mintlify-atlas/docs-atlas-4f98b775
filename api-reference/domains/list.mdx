---
title: "Retrieve a list of domains"
description: "Retrieve a list of domains associated with the authenticated workspace"
---


## Authentication

This endpoint requires authentication using a Bearer token.

<CodeGroup>
```bash cURL
curl -X GET "https://api.dub.co/domains?page=1&pageSize=50" \
  -H "Authorization: Bearer <token>"
```

```typescript TypeScript
import { Dub } from 'dub';

const dub = new Dub({ token: process.env.DUB_API_KEY });

const domains = await dub.domains.list({
  page: 1,
  pageSize: 50,
});
```

```python Python
from dub import Dub

dub = Dub(token=os.environ['DUB_API_KEY'])

domains = dub.domains.list(
  page=1,
  page_size=50,
)
```
</CodeGroup>

## Query Parameters

<ParamField query="archived" type="boolean" default="false">
  Whether to include archived domains in the response. Defaults to `false` if not provided.
</ParamField>

<ParamField query="search" type="string">
  The search term to filter the domains by.
</ParamField>

<ParamField query="page" type="number" default="1">
  The page number for pagination.
</ParamField>

<ParamField query="pageSize" type="number" default="50">
  The number of items per page. Maximum: 50
</ParamField>

## Response

Returns an array of domain objects.

<ResponseField name="id" type="string">
  The unique identifier of the domain.
</ResponseField>

<ResponseField name="slug" type="string">
  The domain name.
  
  Example: `acme.com`
</ResponseField>

<ResponseField name="verified" type="boolean">
  Whether the domain is verified.
</ResponseField>

<ResponseField name="primary" type="boolean">
  Whether the domain is the primary domain for the workspace.
</ResponseField>

<ResponseField name="archived" type="boolean">
  Whether the domain is archived.
</ResponseField>

<ResponseField name="placeholder" type="string | null">
  Provide context to your teammates in the link creation modal by showing them an example of a link to be shortened.
</ResponseField>

<ResponseField name="expiredUrl" type="string | null">
  The URL to redirect to when a link under this domain has expired.
</ResponseField>

<ResponseField name="notFoundUrl" type="string | null">
  The URL to redirect to when a link under this domain doesn't exist.
</ResponseField>

<ResponseField name="logo" type="string | null">
  The logo of the domain.
</ResponseField>

<ResponseField name="assetLinks" type="string | null">
  assetLinks.json configuration file (for deep link support on Android).
</ResponseField>

<ResponseField name="appleAppSiteAssociation" type="string | null">
  apple-app-site-association configuration file (for deep link support on iOS).
</ResponseField>

<ResponseField name="createdAt" type="date">
  The date the domain was created.
</ResponseField>

<ResponseField name="updatedAt" type="date">
  The date the domain was last updated.
</ResponseField>

<ResponseField name="registeredDomain" type="object | null">
  The registered domain record.
  
  <ResponseField name="id" type="string">
    The ID of the registered domain record.
  </ResponseField>
  
  <ResponseField name="autoRenewalDisabledAt" type="date | null">
    The date the domain auto-renew is disabled.
  </ResponseField>
  
  <ResponseField name="createdAt" type="date">
    The date the domain was created.
  </ResponseField>
  
  <ResponseField name="expiresAt" type="date">
    The date the domain expires.
  </ResponseField>
  
  <ResponseField name="renewalFee" type="number">
    The fee to renew the domain.
  </ResponseField>
</ResponseField>
